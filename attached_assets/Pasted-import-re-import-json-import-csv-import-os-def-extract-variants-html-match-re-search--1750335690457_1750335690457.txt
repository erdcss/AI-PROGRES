import re
import json
import csv
import os

def extract_variants(html):
    match = re.search(r'window\.TYPageData\s*=\s*({.*?});', html, re.DOTALL)
    if not match:
        print("Varyant verisi bulunamadı.")
        return []

    try:
        data = json.loads(match.group(1))
        product = data.get("product", {})
        variants = product.get("variants", [])
        imageMap = product.get("variantImageMap", {})  # color -> image
        return variants, imageMap
    except json.JSONDecodeError:
        print("JSON parse hatası.")
        return [], {}

def save_as_json(variants, filename="variants.json"):
    with open(filename, "w", encoding="utf-8") as f:
        json.dump(variants, f, ensure_ascii=False, indent=2)
    print(f"✅ JSON çıktı kaydedildi: {filename}")

def save_as_csv(variants, image_map, filename="variants.csv"):
    os.makedirs(os.path.dirname(filename), exist_ok=True)

    with open(filename, mode="w", newline="", encoding="utf-8") as csvfile:
        fieldnames = ["Renk", "Beden", "Fiyat (TL)", "Stok", "Görsel"]
        writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
        writer.writeheader()

        for var in variants:
            color = var.get("color", "-")
            size = var.get("size", "-")
            price = var.get("price", "-")
            stock = var.get("stock", False)

            stock_display = "✔️ Var" if stock else "❌ Yok"
            image_url = image_map.get(color.lower()) if image_map else None
            if isinstance(image_url, list):
                image_url = image_url[0]  # sadece ilk görseli al

            writer.writerow({
                "Renk": color,
                "Beden": size,
                "Fiyat (TL)": price,
                "Stok": stock_display,
                "Görsel": f"https://cdn.dsmcdn.com{image_url}" if image_url else "-"
            })

    print(f"✅ CSV çıktı kaydedildi: {filename}")

# 🧪 Örnek kullanım
if __name__ == "__main__":
    with open("trendyol_sayfa_kaynagi.html", "r", encoding="utf-8") as f:
        html = f.read()

    variants, image_map = extract_variants(html)

    # JSON ve CSV olarak kaydet
    save_as_json(variants, "output/variants.json")
    save_as_csv(variants, image_map, "output/variants.csv")
