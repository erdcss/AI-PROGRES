import re
import json
import csv
import os
import unicodedata

def slugify(value):
    value = unicodedata.normalize('NFKD', value).encode('ascii', 'ignore').decode('ascii')
    return re.sub(r'[^a-z0-9]+', '-', value.lower()).strip('-')

def extract_variants(html):
    match = re.search(r'window\.TYPageData\s*=\s*({.*?});', html, re.DOTALL)
    if not match:
        print("Varyant verisi bulunamadı.")
        return [], {}

    try:
        data = json.loads(match.group(1))
        product = data.get("product", {})
        variants = product.get("variants", [])
        imageMap = product.get("variantImageMap", {})  # color -> image[]
        title = product.get("name", "Ürün")
        return variants, imageMap, title
    except json.JSONDecodeError:
        print("JSON parse hatası.")
        return [], {}, "Ürün"

def save_shopify_csv(variants, image_map, product_title, filename="output/shopify-variants.csv"):
    os.makedirs(os.path.dirname(filename), exist_ok=True)

    with open(filename, mode="w", newline="", encoding="utf-8-sig") as csvfile:
        fieldnames = [
            "Handle", "Title", "Option1 Name", "Option1 Value",
            "Option2 Name", "Option2 Value", "Variant Price", "Image Src"
        ]
        writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
        writer.writeheader()

        handle = slugify(product_title)

        for v in variants:
            color = v.get("color", "-")
            size = v.get("size", "-")
            price = v.get("price", "0.00")
            image_url = image_map.get(color.lower()) if image_map else None
            if isinstance(image_url, list):
                image_url = image_url[0]
            full_image_url = f"https://cdn.dsmcdn.com{image_url}" if image_url else ""

            writer.writerow({
                "Handle": handle,
                "Title": f"{product_title} - {color} {size}",
                "Option1 Name": "Renk",
                "Option1 Value": color,
                "Option2 Name": "Beden",
                "Option2 Value": size,
                "Variant Price": price,
                "Image Src": full_image_url
            })

    print(f"✅ Shopify CSV kaydedildi: {filename}")

# 🧪 Kullanım
if __name__ == "__main__":
    with open("trendyol_sayfa_kaynagi.html", "r", encoding="utf-8") as f:
        html = f.read()

    variants, image_map, product_title = extract_variants(html)
    save_shopify_csv(variants, image_map, product_title)
